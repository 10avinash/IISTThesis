\section{Including Algorithms}\label{sec:algorithms}
\begin{algorithm}[h]
	\caption{Sum of first $N$ natural numbers}
	\begin{algorithmic}[1]
		\State $S \leftarrow 0$
		\For {$i=1$ to $N$}
			\State $ S \leftarrow S + i $
		\EndFor{}
		\State Output $S$
	\end{algorithmic}
	\label{alg:sum}
\end{algorithm}
